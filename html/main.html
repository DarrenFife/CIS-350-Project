263    def show_videos(self, organized_search_list):
264        """Show videos on search page when clicking go"""
265        if self._inner_layer2.count() > 0:
266            self.clear_window(self._inner_layer2)
267        for video in organized_search_list[self._page_number]:
268            time = self.get_time(video.length)
269            button = Qtw.QPushButton(f"     {video.title}\n\n     "
270                                     f"Created by: {video.author}\n\n   "
271                                     f"  Length: {time[0]}:{time[1]}")
272            button.setFont(Qtg.QFont("Times", 20))
273            pixmap = Qtg.QPixmap()
274            pixmap.loadFromData(requests.get(video.thumbnail_url,
275                                             timeout=5).content)
276            button.setIcon(Qtg.QIcon(pixmap))
277            button.setIconSize(Qtc.QSize(400, 235))
278            button.setStyleSheet("text-align:left;")
279            url = video.watch_url
280            button.clicked.connect(lambda:
281                                   pytube_code.download_link(url,
282                                                             self._resolution))
283            button.setMinimumSize(0, 250)
284            self._inner_layer2.addWidget(button, 1,
285                                         Qtc.Qt.Alignment(Qtc.Qt.AlignTop))
286        prev_button = Qtw.QPushButton("Previous")
287        prev_button.clicked.connect(self.previous_button)
288        nxt_button = Qtw.QPushButton("Next")
289        nxt_button.clicked.connect(self.next_button)
290        self.clear_window(self._inner_layer3)
291        if 0 < self._page_number < len(organized_search_list) - 1:
292            self._inner_layer3.addWidget(prev_button)
293            self._inner_layer3.addWidget(nxt_button)
294        if self._page_number == 0:
295            self._inner_layer3.addWidget(nxt_button)
296        if self._page_number == len(organized_search_list) - 1:
297            self._inner_layer3.addWidget(prev_button)

352        """Clears the videos from search page
353        once we don't need them anymore"""

357                item.widget().close()

360    def page_view_list(self, lov):
361        """Create sorted list of video objects for show_videos to
362        reference from"""
363        num = 0
364        if len(lov) % 3 == 0:
365            num = len(lov) // 3
366        else:
367            num = (len(lov) // 3) + 1
368        new_lst = [[] for i in range(num)]
369        for i in new_lst:
370            for j in range(3):
371                if len(lov) >= 3:
372                    i.append(lov[j])
373                elif len(lov) > 0:
374                    for x in lov:
375                        i.append(x)
376                    break
377                else:
378                    break
379            lov = lov[3:]
380        return new_lst

393def fetch_updates():
394    """Fetch new videos from channels we are subscribed to"""
395    try:
396        f_hand = open(os.pardir + "/YouTube-Downloads/programInfo.txt", 'r',
397                      encoding="utf-8")
398        program_info = f_hand.readlines()
399    except FileNotFoundError:
400        if not os.path.exists(os.pardir + "/YouTube-Downloads/"):
401            os.makedirs(os.pardir + "/YouTube-Downloads/")
402        f_hand = open(os.pardir + "/YouTube-Downloads/programInfo.txt", 'x',
403                      encoding="utf-8")
404        program_info = [date.today().ctime() + "\n"]
405        f_hand.writelines(program_info)
406    if program_info[0] != date.today().ctime() + "\n":
407        print("Fetching updates!")
408        for i in range(1, len(program_info)):
409            pytube_code.download_link(program_info[i], 720)
410    f_hand.close()
